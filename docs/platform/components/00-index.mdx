## Quick Start

```typescript
import {
  EditorShell,
  EditorToolbar,
  EditorDockLayout,
  EditorDockPanel,
  EditorStatusBar,
} from '@forge/shared/components/editor';

export function MyEditor() {
  return (
    <EditorShell editorId="my-editor" title="My Editor">
      <EditorToolbar>
        <EditorToolbar.Left>
          <Button>New</Button>
        </EditorToolbar.Left>
      </EditorToolbar>

      <EditorDockLayout layoutId="my-layout">
        <EditorDockLayout.Main>
          <EditorDockPanel id="main" title="Viewport">
            <MyContent />
          </EditorDockPanel>
        </EditorDockLayout.Main>
      </EditorDockLayout>

      <EditorStatusBar>Ready</EditorStatusBar>
    </EditorShell>
  );
}
```
