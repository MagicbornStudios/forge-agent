## ğŸš€ Getting Started

### Installation

```bash
# Clone repository
git clone https://github.com/your-org/forge-agent

# Install dependencies
yarn install

# Start development server
yarn dev
```

### Your First Editor

```typescript
import {
  EditorShell,
  EditorToolbar,
  EditorDockLayout,
  EditorDockPanel,
} from '@forge/shared/components/editor';

export function MyEditor() {
  return (
    <EditorShell editorId="my-editor" title="My Editor">
      <EditorToolbar>
        <EditorToolbar.Left>
          <Button>New</Button>
        </EditorToolbar.Left>
      </EditorToolbar>

      <EditorDockLayout layoutId="my-layout">
        <EditorDockLayout.Main>
          <EditorDockPanel id="main" title="Main">
            <div>Your content here</div>
          </EditorDockPanel>
        </EditorDockLayout.Main>
      </EditorDockLayout>
    </EditorShell>
  );
}
```
