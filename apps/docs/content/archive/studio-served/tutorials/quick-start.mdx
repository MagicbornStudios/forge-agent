# Quick Start: Build Your First Editor in 10 Minutes

Build a functional editor with the Forge platform from scratch. This tutorial covers the essential components and patterns to get you productive quickly.

## What You'll Build

A simple task management editor with:

- Left sidebar (task list)
- Main viewport (task editor)
- Right inspector (task properties)
- Toolbar with actions
- Status bar
- Modal dialogs

## Prerequisites

```bash
# Install dependencies
npm install @forge/dev-kit
```

## Step 2: Add the Layout (3 minutes)

Replace the placeholder content with `WorkspaceLayout`:

```tsx
'use client';

import { useState } from 'react';
import {
  EditorShell,
  EditorToolbar,
  WorkspaceLayout,
  WorkspacePanel,
  EditorStatusBar,
  EditorOverlaySurface,
} from '@forge/dev-kit';
import { Plus, ListTodo, FileEdit, ScanSearch } from 'lucide-react';
import { Card } from '@forge/dev-kit/ui/card';
import { Button } from '@forge/dev-kit/ui/button';

interface Task {
  id: number;
  title: string;
  description: string;
  completed: boolean;
}

export function TaskEditor() {
  const [tasks, setTasks] = useState<Task[]>([
    { id: 1, title: 'Build editor', description: 'Create task editor', completed: false },
    { id: 2, title: 'Add features', description: 'Add CRUD operations', completed: false },
  ]);
  const [activeTaskId, setActiveTaskId] = useState<number | null>(1);

  const activeTask = tasks.find((t) => t.id === activeTaskId);

  return (
    <EditorShell
      editorId="tasks"
      title="Tasks"
      domain="tasks"
      className="bg-canvas"
    >
      <EditorToolbar className="bg-sidebar border-b">
        <EditorToolbar.Left>
          <span className="text-xs font-medium">{tasks.length} tasks</span>
        </EditorToolbar.Left>
        <EditorToolbar.Right>
          <EditorToolbar.Button
            size="sm"
            variant="outline"
            onClick={() => console.log('Create task')}
          >
            <Plus size={14} />
            New Task
          </EditorToolbar.Button>
        </EditorToolbar.Right>
      </EditorToolbar>

      <WorkspaceLayout
        layoutId="tasks-editor"
        leftDefaultSize={20}
        rightDefaultSize={25}
      >
        {/* Left: Task List */}
        <WorkspaceLayout.Left>
          <WorkspaceLayout.Panel id="list" title="Tasks" icon={<ListTodo size={14} />}>
            <WorkspacePanel panelId="task-list" hideTitleBar scrollable>
              <div className="space-y-1">
                {tasks.map((task) => (
                  <Button
                    key={task.id}
                    variant="ghost"
                    className="w-full justify-start text-left"
                    onClick={() => setActiveTaskId(task.id)}
                    data-state={activeTaskId === task.id ? 'active' : undefined}
                  >
                    <span className="truncate">{task.title}</span>
                  </Button>
                ))}
              </div>
            </WorkspacePanel>
          </WorkspaceLayout.Panel>
        </WorkspaceLayout.Left>

        {/* Main: Task Editor */}
        <WorkspaceLayout.Main>
          <WorkspaceLayout.Panel id="editor" title="Editor" icon={<FileEdit size={14} />}>
            <WorkspacePanel panelId="task-editor" hideTitleBar scrollable={false}>
              {activeTask ? (
                <div className="flex flex-col h-full p-4">
                  <Card className="p-4">
                    <h2 className="text-lg font-semibold mb-2">{activeTask.title}</h2>
                    <p className="text-sm text-muted-foreground">{activeTask.description}</p>
                  </Card>
                </div>
              ) : (
                <div className="flex items-center justify-center h-full">
                  <p className="text-muted-foreground">Select a task</p>
                </div>
              )}
            </WorkspacePanel>
          </WorkspaceLayout.Panel>
        </WorkspaceLayout.Main>

        {/* Right: Properties */}
        <WorkspaceLayout.Right>
          <WorkspaceLayout.Panel id="props" title="Properties" icon={<ScanSearch size={14} />}>
            <WorkspacePanel panelId="task-props" hideTitleBar scrollable>
              {activeTask ? (
                <div className="space-y-4">
                  <div>
                    <label className="text-xs font-medium text-muted-foreground">ID</label>
                    <p className="text-sm">{activeTask.id}</p>
                  </div>
                  <div>
                    <label className="text-xs font-medium text-muted-foreground">Status</label>
                    <p className="text-sm">{activeTask.completed ? 'Completed' : 'Pending'}</p>
                  </div>
                </div>
              ) : (
                <p className="text-xs text-muted-foreground">No task selected</p>
              )}
            </WorkspacePanel>
          </WorkspaceLayout.Panel>
        </WorkspaceLayout.Right>
      </WorkspaceLayout>

      <EditorStatusBar>
        {activeTask ? `Editing: ${activeTask.title}` : 'Ready'}
      </EditorStatusBar>
    </EditorShell>
  );
}
```

**What you got:**

- 3-pane layout (left | main | right)
- Resizable panels
- Task list on left
- Task editor in main
- Properties on right
- Active task highlighting
- Layout persistence (via `layoutId`)

## Step 3: Add CRUD Operations (3 minutes)

Add create, update, and delete operations:

```tsx
'use client';

import { useState, useCallback } from 'react';
import {
  EditorShell,
  EditorToolbar,
  WorkspaceLayout,
  WorkspacePanel,
  EditorStatusBar,
  EditorOverlaySurface,
} from '@forge/dev-kit';
import type { OverlaySpec, ActiveOverlay } from '@forge/dev-kit';
import { Plus, ListTodo, FileEdit, ScanSearch, Trash } from 'lucide-react';
import { Card } from '@forge/dev-kit/ui/card';
import { Button } from '@forge/dev-kit/ui/button';
import { Input } from '@forge/dev-kit/ui/input';
import { Textarea } from '@forge/dev-kit/ui/textarea';

interface Task {
  id: number;
  title: string;
  description: string;
  completed: boolean;
}

export function TaskEditor() {
  const [tasks, setTasks] = useState<Task[]>([
    { id: 1, title: 'Build editor', description: 'Create task editor', completed: false },
    { id: 2, title: 'Add features', description: 'Add CRUD operations', completed: false },
  ]);
  const [activeTaskId, setActiveTaskId] = useState<number | null>(1);
  const [activeOverlay, setActiveOverlay] = useState<ActiveOverlay | null>(null);

  const activeTask = tasks.find((t) => t.id === activeTaskId);

  const handleCreateTask = useCallback((title: string, description: string) => {
    const newTask: Task = {
      id: Date.now(),
      title,
      description,
      completed: false,
    };
    setTasks((prev) => [...prev, newTask]);
    setActiveTaskId(newTask.id);
    setActiveOverlay(null);
  }, []);

  const handleUpdateTask = useCallback(
    (id: number, updates: Partial<Task>) => {
      setTasks((prev) =>
        prev.map((t) => (t.id === id ? { ...t, ...updates } : t))
      );
    },
    []
  );

  const handleDeleteTask = useCallback((id: number) => {
    setTasks((prev) => prev.filter((t) => t.id !== id));
    if (activeTaskId === id) {
      setActiveTaskId(tasks[0]?.id ?? null);
    }
  }, [activeTaskId, tasks]);

  const overlays: OverlaySpec[] = [
    {
      id: 'create-task',
      type: 'modal',
      title: 'Create Task',
      size: 'md',
      render: ({ onDismiss }) => <CreateTaskForm onCreate={handleCreateTask} onClose={onDismiss} />,
    },
  ];

  return (
    <EditorShell
      editorId="tasks"
      title="Tasks"
      domain="tasks"
      className="bg-canvas"
    >
      <EditorToolbar className="bg-sidebar border-b">
        <EditorToolbar.Left>
          <span className="text-xs font-medium">{tasks.length} tasks</span>
        </EditorToolbar.Left>
        <EditorToolbar.Right>
          <EditorToolbar.Button
            size="sm"
            variant="outline"
            onClick={() => setActiveOverlay({ id: 'create-task' })}
          >
            <Plus size={14} />
            New Task
          </EditorToolbar.Button>
        </EditorToolbar.Right>
      </EditorToolbar>

      <WorkspaceLayout
        layoutId="tasks-editor"
        leftDefaultSize={20}
        rightDefaultSize={25}
      >
        {/* Left: Task List */}
        <WorkspaceLayout.Left>
          <WorkspaceLayout.Panel id="list" title="Tasks" icon={<ListTodo size={14} />}>
            <WorkspacePanel panelId="task-list" hideTitleBar scrollable>
              <div className="space-y-1">
                {tasks.map((task) => (
                  <div key={task.id} className="flex items-center gap-1">
                    <Button
                      variant="ghost"
                      className="flex-1 justify-start text-left"
                      onClick={() => setActiveTaskId(task.id)}
                      data-state={activeTaskId === task.id ? 'active' : undefined}
                    >
                      <span className="truncate">{task.title}</span>
                    </Button>
                    <Button
                      size="sm"
                      variant="ghost"
                      onClick={() => handleDeleteTask(task.id)}
                    >
                      <Trash size={12} />
                    </Button>
                  </div>
                ))}
              </div>
            </WorkspacePanel>
          </WorkspaceLayout.Panel>
        </WorkspaceLayout.Left>

        {/* Main: Task Editor */}
        <WorkspaceLayout.Main>
          <WorkspaceLayout.Panel id="editor" title="Editor" icon={<FileEdit size={14} />}>
            <WorkspacePanel panelId="task-editor" hideTitleBar scrollable>
              {activeTask ? (
                <div className="space-y-4">
                  <div>
                    <label className="text-sm font-medium">Title</label>
                    <Input
                      value={activeTask.title}
                      onChange={(e) =>
                        handleUpdateTask(activeTask.id, { title: e.target.value })
                      }
                    />
                  </div>
                  <div>
                    <label className="text-sm font-medium">Description</label>
                    <Textarea
                      value={activeTask.description}
                      onChange={(e) =>
                        handleUpdateTask(activeTask.id, { description: e.target.value })
                      }
                      rows={5}
                    />
                  </div>
                  <div className="flex items-center gap-2">
                    <input
                      type="checkbox"
                      checked={activeTask.completed}
                      onChange={(e) =>
                        handleUpdateTask(activeTask.id, { completed: e.target.checked })
                      }
                    />
                    <label className="text-sm">Completed</label>
                  </div>
                </div>
              ) : (
                <div className="flex items-center justify-center h-full">
                  <p className="text-muted-foreground">Select a task</p>
                </div>
              )}
            </WorkspacePanel>
          </WorkspaceLayout.Panel>
        </WorkspaceLayout.Main>

        {/* Right: Properties */}
        <WorkspaceLayout.Right>
          <WorkspaceLayout.Panel id="props" title="Properties" icon={<ScanSearch size={14} />}>
            <WorkspacePanel panelId="task-props" hideTitleBar scrollable>
              {activeTask ? (
                <div className="space-y-4">
                  <div>
                    <label className="text-xs font-medium text-muted-foreground">ID</label>
                    <p className="text-sm">{activeTask.id}</p>
                  </div>
                  <div>
                    <label className="text-xs font-medium text-muted-foreground">Status</label>
                    <p className="text-sm">{activeTask.completed ? 'Completed' : 'Pending'}</p>
                  </div>
                  <div>
                    <label className="text-xs font-medium text-muted-foreground">Characters</label>
                    <p className="text-sm">{activeTask.description.length}</p>
                  </div>
                </div>
              ) : (
                <p className="text-xs text-muted-foreground">No task selected</p>
              )}
            </WorkspacePanel>
          </WorkspaceLayout.Panel>
        </WorkspaceLayout.Right>
      </WorkspaceLayout>

      <EditorStatusBar>
        {activeTask ? `Editing: ${activeTask.title}` : 'Ready'}
      </EditorStatusBar>

      <EditorOverlaySurface
        overlays={overlays}
        activeOverlay={activeOverlay}
        onDismiss={() => setActiveOverlay(null)}
      />
    </EditorShell>
  );
}

// Create Task Form Component
function CreateTaskForm({
  onCreate,
  onClose,
}: {
  onCreate: (title: string, description: string) => void;
  onClose: () => void;
}) {
  const [title, setTitle] = useState('');
  const [description, setDescription] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (title.trim()) {
      onCreate(title, description);
      onClose();
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div>
        <label className="text-sm font-medium">Title</label>
        <Input
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          placeholder="Enter task title"
          autoFocus
        />
      </div>
      <div>
        <label className="text-sm font-medium">Description</label>
        <Textarea
          value={description}
          onChange={(e) => setDescription(e.target.value)}
          placeholder="Enter task description"
          rows={4}
        />
      </div>
      <div className="flex justify-end gap-2">
        <Button type="button" variant="outline" onClick={onClose}>
          Cancel
        </Button>
        <Button type="submit" disabled={!title.trim()}>
          Create
        </Button>
      </div>
    </form>
  );
}
```

**What you got:**

- Create task modal
- Edit task (inline editing)
- Delete task
- Real-time status updates
- Form validation

## Step 4: Add Settings Integration (2 minutes)

Connect to the settings system:

```tsx
import { useSettingsStore } from '@/lib/settings/store';

export function TaskEditor() {
  const editorId = 'tasks';

  // Pull theme and density from settings
  const editorTheme = useSettingsStore((s) =>
    s.getSettingValue('ui.theme', { editorId })
  ) as string | undefined;

  const editorDensity = useSettingsStore((s) =>
    s.getSettingValue('ui.density', { editorId })
  ) as string | undefined;

  // Pull panel visibility from settings
  const showLeftPanel = useSettingsStore((s) =>
    s.getSettingValue('panel.visible.tasks-left', { editorId })
  ) as boolean | undefined;

  const showRightPanel = useSettingsStore((s) =>
    s.getSettingValue('panel.visible.tasks-right', { editorId })
  ) as boolean | undefined;

  // ... rest of state

  return (
    <EditorShell
      editorId={editorId}
      title="Tasks"
      domain="tasks"
      theme={editorTheme}
      density={editorDensity}
      className="bg-canvas"
    >
      {/* ... toolbar */}

      <WorkspaceLayout layoutId="tasks-editor">
        {showLeftPanel !== false && (
          <WorkspaceLayout.Left>
            {/* ... left panel */}
          </WorkspaceLayout.Left>
        )}

        <WorkspaceLayout.Main>
          {/* ... main panel */}
        </WorkspaceLayout.Main>

        {showRightPanel !== false && (
          <WorkspaceLayout.Right>
            {/* ... right panel */}
          </WorkspaceLayout.Right>
        )}
      </WorkspaceLayout>

      {/* ... status bar, overlays */}
    </EditorShell>
  );
}
```

**What you got:**

- Theme switching (dark/light)
- Density adjustment (compact/comfortable)
- Panel visibility toggle
- Settings persistence
